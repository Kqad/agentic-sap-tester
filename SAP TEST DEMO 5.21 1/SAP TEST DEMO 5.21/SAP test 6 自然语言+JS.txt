async function run(agent) {
  await agent.aiTap("进入 Menu");
  await agent.aiTap("点击setting");
  await agent.aiTap("点击visualization");
  await agent.aiTap("启用show Ok code field");
  await agent.aiTap("保存");
  await agent.aiInput("左上角矩形", { value: "FAGLL03H" });
  await agent.aiTap("点击 Execute 按钮执行查询");
  await agent.aiInput("company code", { value: "8540" });
  await agent.aiTap("点击all items");
  await agent.aiInput("posting date开始日期", { value: "01.12.2025" });
  await agent.aiInput("posting date截至日期", { value: "31.12.2025" });
  await agent.aiTap("点击 Execute 按钮执行查询");
  await sleep(4000);
  await agent.aiTap("点击图标", { xpath: "//*[@id=\"C185_toolbar_btn10\"]" });
  await sleep(2000);
  await agent.aiTap("点击右边栏目column set下面的搜索");
  await agent.aiInput("search term", { value: "operational division" });
  await agent.aiTap("点击ok");
  await agent.aiTap("点击×");
  await agent.aiTap("点击向左的箭头");
  await agent.aiTap("点击adopt");
  await sleep(6000);
  await agent.aiAct("按住底部的横向滚动条的滑块，拖到最右端");
  await agent.aiAssert("页面上能看到oper. div.即为测试成功");
}
